CREATE TABLE userdata(id NUMBER(10) NOT NULL,email VARCHAR2(255),first_name  VARCHAR2(255),last_name VARCHAR2(255),PRIMARY KEY (id));

CREATE SEQUENCE userdata_seq START WITH 1 INCREMENT BY 1; 

CREATE OR REPLACE TRIGGER userdata_seq_tr BEFORE INSERT ON userdata FOR EACH ROW WHEN (NEW.id IS NULL) BEGIN SELECT userdata_seq.NEXTVAL INTO :NEW.id FROM DUAL; END;
